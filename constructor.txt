# CONSTRUCTOR
class Example{

Example(){
System.out.println(&quot;Default constructor&quot;);
}

Example(int i, int j){
System.out.println(&quot;constructor with Two parameters&quot;);
}

Example(int i, int j, int k){
System.out.println(&quot;constructor with Three parameters&quot;);
}

Example(int i, String name){
System.out.println(&quot;constructor with int and String param&quot;);
}
public static void main(String args[]){

Example obj = new Example();

Example obj2 = new Example(12, 12);

Example obj3 = new Example(1, 2, 13);

Example obj4 = new Example(1,&quot;BeginnersBook&quot;);

}
}
